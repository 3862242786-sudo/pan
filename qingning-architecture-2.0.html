// View Transitions API - 平滑页面切换
async function switchPage(targetUrl) {
    // 检查浏览器支持
    if (!document.startViewTransition) {
        window.location.href = targetUrl;
        return;
    }

    // 启动视图过渡
    const transition = document.startViewTransition(async () => {
        // 更新 DOM 内容
        const response = await fetch(targetUrl);
        const html = await response.text();
        const parser = new DOMParser();
        const doc = parser.parseFromString(html, 'text/html');
        document.body.innerHTML = doc.body.innerHTML;
    });

    // 等待过渡完成
    await transition.finished;
}// CSS 自定义过渡动画
::view-transition-old(root) {
    animation: fadeOut 0.3s ease;
}
::view-transition-new(root) {
    animation: fadeIn 0.3s ease;
}

// main.js
const canvas = document.getElementById('game');
const offscreen = canvas.transferControlToOffscreen();

const worker = new Worker('qge-worker.js');
worker.postMessage({
    type: 'init',
    canvas: offscreen,
    width: 800,
    height: 600
}, [offscreen]);

// qge-worker.js
self.onmessage = function(e) {
    const { type, canvas, width, height } = e.data;
    if (type === 'init') {
        // 在 Worker 中初始化 QGE
        const game = new QGE.Core({
            canvas: canvas,
            width: width,
            height: height,
            offscreen: true
        });
        game.start();
    }
};

// 主线程发送输入事件到 Worker
canvas.addEventListener('touchstart', (e) => {
    worker.postMessage({
        type: 'input',
        event: 'touchstart',
        touches: [...e.touches].map(t => ({ x: t.clientX, y: t.clientY }))
    });
});
// OPFS - 高速游戏存档系统
class GameSaveManager {
    constructor() {
        this.root = null;
    }

    async init() {
        // 获取 OPFS 根目录句柄
        this.root = await navigator.storage.getDirectory();
    }

    async saveGame(slotId, data) {
        // 创建/打开存档文件
        const fileHandle = await this.root.getFileHandle(
            `save_${slotId}.json`,
            { create: true }
        );

        // 同步写入 (性能更高)
        const writable = await fileHandle.createSyncAccessHandle();
        const encoder = new TextEncoder();
        const buffer = encoder.encode(JSON.stringify(data));
        writable.write(buffer);
        writable.close();
    }

    async loadGame(slotId) {
        try {
            const fileHandle = await this.root.getFileHandle(`save_${slotId}.json`);
            const file = await fileHandle.getFile();
            const text = await file.text();
            return JSON.parse(text);
        } catch (e) {
            return null;
        }
    }

    async listSaves() {
        const saves = [];
        for await (const [name, handle] of this.root.entries()) {
            if (name.startsWith('save_')) {
                saves.push(name);
            }
        }
        return saves;
    }
}
// QNA API 集成示例
class QNAClient {
    constructor(baseURL) {
        this.baseURL = baseURL;
        this.token = localStorage.getItem('qna_token');
    }

    async request(endpoint, options = {}) {
        const url = `${this.baseURL}/api/v1${endpoint}`;
        const headers = {
            'Content-Type': 'application/json',
            ...options.headers
        };
        if (this.token) {
            headers['Authorization'] = `Bearer ${this.token}`;
        }

        const response = await fetch(url, { ...options, headers });
        if (!response.ok) {
            throw new Error(`API Error: ${response.status}`);
        }
        return response.json();
    }

    // 用户认证
    async login(username, password) {
        const data = await this.request('/auth/login', {
            method: 'POST',
            body: JSON.stringify({ username, password })
        });
        this.token = data.token;
        localStorage.setItem('qna_token', data.token);
        return data;
    }

    // 游戏数据同步
    async syncGameData(gameData) {
        return this.request('/game/sync', {
            method: 'POST',
            body: JSON.stringify({
                timestamp: Date.now(),
                data: gameData,
                checksum: await this.computeChecksum(gameData)
            })
        });
    }

    // 实时同步 - WebSocket
    connectRealtime(roomId) {
        const ws = new WebSocket(
            `wss://${this.baseURL}/ws/room/${roomId}?token=${this.token}`
        );
        ws.onmessage = (event) => {
            const msg = JSON.parse(event.data);
            this.emit('sync', msg);
        };
        return ws;
    }
}
